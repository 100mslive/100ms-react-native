# react-native-hmssdk

React native wrapper for 100ms SDK

## Installation

```sh
npm install react-native-hmssdk
```

## Usage

```js
import Hmssdk from "react-native-hmssdk";

// ...

<HmssdkViewManager
  color="#32a852"
  userId={userId}
  roomId={roomId}
  authToken={token}
  style={styles.box}
  layout={{ width, height }}
  isMute={isMute}
  switchCamera={switchCamera}
  muteVideo={muteVideo}
/>

```
